274 template<typename XprType> struct blas_traits
275 {
276  typedef typename traits<XprType>::Scalar Scalar;
277  typedef const XprType& ExtractType;
278  typedef XprType _ExtractType;
279  enum {
280  IsComplex = NumTraits<Scalar>::IsComplex,
281  IsTransposed = false,
282  NeedToConjugate = false,
283  HasUsableDirectAccess = ( (int(XprType::Flags)&DirectAccessBit)
284  && ( bool(XprType::IsVectorAtCompileTime)
285  || int(inner_stride_at_compile_time<XprType>::ret) == 1)
286  ) ? 1 : 0
287  };
288  typedef typename conditional<bool(HasUsableDirectAccess),
289  ExtractType,
290  typename _ExtractType::PlainObject
291  >::type DirectLinearAccessType;
292  static inline ExtractType extract(const XprType& x) { return x; }
293  static inline const Scalar extractScalarFactor(const XprType&) { return Scalar(1); }
294 };
