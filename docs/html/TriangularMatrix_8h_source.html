648 template<typename Derived>
649 bool MatrixBase<Derived>::isUpperTriangular(const RealScalar& prec) const
650 {
651  RealScalar maxAbsOnUpperPart = static_cast<RealScalar>(-1);
652  for(Index j = 0; j < cols(); ++j)
653  {
654  Index maxi = numext::mini(j, rows()-1);
655  for(Index i = 0; i <= maxi; ++i)
656  {
657  RealScalar absValue = numext::abs(coeff(i,j));
658  if(absValue > maxAbsOnUpperPart) maxAbsOnUpperPart = absValue;
659  }
660  }
661  RealScalar threshold = maxAbsOnUpperPart * prec;
662  for(Index j = 0; j < cols(); ++j)
663  for(Index i = j+1; i < rows(); ++i)
664  if(numext::abs(coeff(i, j)) > threshold) return false;
665  return true;
666 }

673 template<typename Derived>
674 bool MatrixBase<Derived>::isLowerTriangular(const RealScalar& prec) const
675 {
676  RealScalar maxAbsOnLowerPart = static_cast<RealScalar>(-1);
677  for(Index j = 0; j < cols(); ++j)
678  for(Index i = j; i < rows(); ++i)
679  {
680  RealScalar absValue = numext::abs(coeff(i,j));
681  if(absValue > maxAbsOnLowerPart) maxAbsOnLowerPart = absValue;
682  }
683  RealScalar threshold = maxAbsOnLowerPart * prec;
684  for(Index j = 1; j < cols(); ++j)
685  {
686  Index maxi = numext::mini(j, rows()-1);
687  for(Index i = 0; i < maxi; ++i)
688  if(numext::abs(coeff(i, j)) > threshold) return false;
689  }
690  return true;
