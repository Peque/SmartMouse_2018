542  inline void Correct() {
543  // In the case sec and nsec have different signs, normalize
544  if (this->sec > 0 && this->nsec < 0) {
545  int32_t n = abs(this->nsec / this->nsInSec) + 1;
546  this->sec -= n;
547  this->nsec += n * this->nsInSec;
548  }
549  if (this->sec < 0 && this->nsec > 0) {
550  int32_t n = abs(this->nsec / this->nsInSec) + 1;
551  this->sec += n;
552  this->nsec -= n * this->nsInSec;
553  }
554 
555  // Make any corrections
556  this->sec += this->nsec / this->nsInSec;
557  this->nsec = this->nsec % this->nsInSec;
558  }
