92 template<typename Scalar>
93 bool JacobiRotation<Scalar>::makeJacobi(const RealScalar& x, const Scalar& y, const RealScalar& z)
94 {
95  using std::sqrt;
96  using std::abs;
97  typedef typename NumTraits<Scalar>::Real RealScalar;
98  RealScalar deno = RealScalar(2)*abs(y);
99  if(deno < (std::numeric_limits<RealScalar>::min)())
100  {
101  m_c = Scalar(1);
102  m_s = Scalar(0);
103  return false;
104  }
105  else
106  {
107  RealScalar tau = (x-z)/deno;
108  RealScalar w = sqrt(numext::abs2(tau) + RealScalar(1));
109  RealScalar t;
110  if(tau>RealScalar(0))
111  {
112  t = RealScalar(1) / (tau + w);
113  }
114  else
115  {
116  t = RealScalar(1) / (tau - w);
117  }
118  RealScalar sign_t = t > RealScalar(0) ? RealScalar(1) : RealScalar(-1);
119  RealScalar n = RealScalar(1) / sqrt(numext::abs2(t)+RealScalar(1));
120  m_s = - sign_t * (numext::conj(y) / abs(y)) * abs(t) * n;
121  m_c = n;
122  return true;
123  }
124 }

166 template<typename Scalar>
167 void JacobiRotation<Scalar>::makeGivens(const Scalar& p, const Scalar& q, Scalar* r, internal::true_type)
168 {
169  using std::sqrt;
170  using std::abs;
171  using numext::conj;
172 
173  if(q==Scalar(0))
174  {
175  m_c = numext::real(p)<0 ? Scalar(-1) : Scalar(1);
176  m_s = 0;
177  if(r) *r = m_c * p;
178  }
179  else if(p==Scalar(0))
180  {
181  m_c = 0;
182  m_s = -q/abs(q);
183  if(r) *r = abs(q);
184  }
185  else
186  {
187  RealScalar p1 = numext::norm1(p);
188  RealScalar q1 = numext::norm1(q);
189  if(p1>=q1)
190  {
191  Scalar ps = p / p1;
192  RealScalar p2 = numext::abs2(ps);
193  Scalar qs = q / p1;
194  RealScalar q2 = numext::abs2(qs);
195 
196  RealScalar u = sqrt(RealScalar(1) + q2/p2);
197  if(numext::real(p)<RealScalar(0))
198  u = -u;
199 
200  m_c = Scalar(1)/u;
201  m_s = -qs*conj(ps)*(m_c/p2);
202  if(r) *r = p * u;
203  }
204  else
205  {
206  Scalar ps = p / q1;
207  RealScalar p2 = numext::abs2(ps);
208  Scalar qs = q / q1;
209  RealScalar q2 = numext::abs2(qs);
210 
211  RealScalar u = q1 * sqrt(p2 + q2);
212  if(numext::real(p)<RealScalar(0))
213  u = -u;
214 
215  p1 = abs(p);
216  ps = p/p1;
217  m_c = p1/u;
218  m_s = -conj(ps) * (q/u);
219  if(r) *r = ps * u;
220  }
221  }
222 }

225 template<typename Scalar>
226 void JacobiRotation<Scalar>::makeGivens(const Scalar& p, const Scalar& q, Scalar* r, internal::false_type)
227 {
228  using std::sqrt;
229  using std::abs;
230  if(q==Scalar(0))
231  {
232  m_c = p<Scalar(0) ? Scalar(-1) : Scalar(1);
233  m_s = Scalar(0);
234  if(r) *r = abs(p);
235  }
236  else if(p==Scalar(0))
237  {
238  m_c = Scalar(0);
239  m_s = q<Scalar(0) ? Scalar(1) : Scalar(-1);
240  if(r) *r = abs(q);
241  }
242  else if(abs(p) > abs(q))
243  {
244  Scalar t = q/p;
245  Scalar u = sqrt(Scalar(1) + numext::abs2(t));
246  if(p<Scalar(0))
247  u = -u;
248  m_c = Scalar(1)/u;
249  m_s = -t * m_c;
250  if(r) *r = p * u;
251  }
252  else
253  {
254  Scalar t = p/q;
255  Scalar u = sqrt(Scalar(1) + numext::abs2(t));
256  if(q<Scalar(0))
257  u = -u;
258  m_s = -Scalar(1)/u;
259  m_c = -t * m_s;
260  if(r) *r = q * u;
261  }

315 struct apply_rotation_in_the_plane_selector
316 {
317  static inline void run(Scalar *x, Index incrx, Scalar *y, Index incry, Index size, OtherScalar c, OtherScalar s)
318  {
319  for(Index i=0; i<size; ++i)
320  {
321  Scalar xi = *x;
322  Scalar yi = *y;
323  *x = c * xi + numext::conj(s) * yi;
324  *y = -s * xi + numext::conj(c) * yi;
325  x += incrx;
326  y += incry;
327  }
328  }

331 template<typename Scalar, typename OtherScalar,
332  int SizeAtCompileTime, int MinAlignment>
333 struct apply_rotation_in_the_plane_selector<Scalar,OtherScalar,SizeAtCompileTime,MinAlignment,true /* vectorizable */>
334 {
335  static inline void run(Scalar *x, Index incrx, Scalar *y, Index incry, Index size, OtherScalar c, OtherScalar s)
336  {
337  enum {
338  PacketSize = packet_traits<Scalar>::size,
339  OtherPacketSize = packet_traits<OtherScalar>::size
340  };
341  typedef typename packet_traits<Scalar>::type Packet;
342  typedef typename packet_traits<OtherScalar>::type OtherPacket;
343 
344  /*** dynamic-size vectorized paths ***/
345  if(SizeAtCompileTime == Dynamic && ((incrx==1 && incry==1) || PacketSize == 1))
346  {
347  // both vectors are sequentially stored in memory => vectorization
348  enum { Peeling = 2 };
349 
350  Index alignedStart = internal::first_default_aligned(y, size);
351  Index alignedEnd = alignedStart + ((size-alignedStart)/PacketSize)*PacketSize;
352 
353  const OtherPacket pc = pset1<OtherPacket>(c);
354  const OtherPacket ps = pset1<OtherPacket>(s);
355  conj_helper<OtherPacket,Packet,NumTraits<OtherScalar>::IsComplex,false> pcj;
356  conj_helper<OtherPacket,Packet,false,false> pm;
357 
358  for(Index i=0; i<alignedStart; ++i)
359  {
360  Scalar xi = x[i];
361  Scalar yi = y[i];
362  x[i] = c * xi + numext::conj(s) * yi;
363  y[i] = -s * xi + numext::conj(c) * yi;
364  }
365 
366  Scalar* EIGEN_RESTRICT px = x + alignedStart;
367  Scalar* EIGEN_RESTRICT py = y + alignedStart;
368 
369  if(internal::first_default_aligned(x, size)==alignedStart)
370  {
371  for(Index i=alignedStart; i<alignedEnd; i+=PacketSize)
372  {
373  Packet xi = pload<Packet>(px);
374  Packet yi = pload<Packet>(py);
375  pstore(px, padd(pm.pmul(pc,xi),pcj.pmul(ps,yi)));
376  pstore(py, psub(pcj.pmul(pc,yi),pm.pmul(ps,xi)));
377  px += PacketSize;
378  py += PacketSize;
379  }
380  }
381  else
382  {
383  Index peelingEnd = alignedStart + ((size-alignedStart)/(Peeling*PacketSize))*(Peeling*PacketSize);
384  for(Index i=alignedStart; i<peelingEnd; i+=Peeling*PacketSize)
385  {
386  Packet xi = ploadu<Packet>(px);
387  Packet xi1 = ploadu<Packet>(px+PacketSize);
388  Packet yi = pload <Packet>(py);
389  Packet yi1 = pload <Packet>(py+PacketSize);
390  pstoreu(px, padd(pm.pmul(pc,xi),pcj.pmul(ps,yi)));
391  pstoreu(px+PacketSize, padd(pm.pmul(pc,xi1),pcj.pmul(ps,yi1)));
392  pstore (py, psub(pcj.pmul(pc,yi),pm.pmul(ps,xi)));
393  pstore (py+PacketSize, psub(pcj.pmul(pc,yi1),pm.pmul(ps,xi1)));
394  px += Peeling*PacketSize;
395  py += Peeling*PacketSize;
396  }
397  if(alignedEnd!=peelingEnd)
398  {
399  Packet xi = ploadu<Packet>(x+peelingEnd);
400  Packet yi = pload <Packet>(y+peelingEnd);
401  pstoreu(x+peelingEnd, padd(pm.pmul(pc,xi),pcj.pmul(ps,yi)));
402  pstore (y+peelingEnd, psub(pcj.pmul(pc,yi),pm.pmul(ps,xi)));
403  }
404  }
405 
406  for(Index i=alignedEnd; i<size; ++i)
407  {
408  Scalar xi = x[i];
409  Scalar yi = y[i];
410  x[i] = c * xi + numext::conj(s) * yi;
411  y[i] = -s * xi + numext::conj(c) * yi;
412  }
413  }
414 
415  /*** fixed-size vectorized path ***/
416  else if(SizeAtCompileTime != Dynamic && MinAlignment>0) // FIXME should be compared to the required alignment
417  {
418  const OtherPacket pc = pset1<OtherPacket>(c);
419  const OtherPacket ps = pset1<OtherPacket>(s);
420  conj_helper<OtherPacket,Packet,NumTraits<OtherPacket>::IsComplex,false> pcj;
421  conj_helper<OtherPacket,Packet,false,false> pm;
422  Scalar* EIGEN_RESTRICT px = x;
423  Scalar* EIGEN_RESTRICT py = y;
424  for(Index i=0; i<size; i+=PacketSize)
425  {
426  Packet xi = pload<Packet>(px);
427  Packet yi = pload<Packet>(py);
428  pstore(px, padd(pm.pmul(pc,xi),pcj.pmul(ps,yi)));
429  pstore(py, psub(pcj.pmul(pc,yi),pm.pmul(ps,xi)));
430  px += PacketSize;
431  py += PacketSize;
432  }
433  }
434 
435  /*** non-vectorized path ***/
436  else
437  {
438  apply_rotation_in_the_plane_selector<Scalar,OtherScalar,SizeAtCompileTime,MinAlignment,false>::run(x,incrx,y,incry,size,c,s);
439  }
440  }
441 };
