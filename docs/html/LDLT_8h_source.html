292  template<typename MatrixType, typename TranspositionType, typename Workspace>
293  static bool unblocked(MatrixType& mat, TranspositionType& transpositions, Workspace& temp, SignMatrix& sign)
294  {
295  using std::abs;
296  typedef typename MatrixType::Scalar Scalar;
297  typedef typename MatrixType::RealScalar RealScalar;
298  typedef typename TranspositionType::StorageIndex IndexType;
299  eigen_assert(mat.rows()==mat.cols());
300  const Index size = mat.rows();
301  bool found_zero_pivot = false;
302  bool ret = true;
303 
304  if (size <= 1)
305  {
306  transpositions.setIdentity();
307  if (numext::real(mat.coeff(0,0)) > static_cast<RealScalar>(0) ) sign = PositiveSemiDef;
308  else if (numext::real(mat.coeff(0,0)) < static_cast<RealScalar>(0)) sign = NegativeSemiDef;
309  else sign = ZeroSign;
310  return true;
311  }
312 
313  for (Index k = 0; k < size; ++k)
314  {
315  // Find largest diagonal element
316  Index index_of_biggest_in_corner;
317  mat.diagonal().tail(size-k).cwiseAbs().maxCoeff(&index_of_biggest_in_corner);
318  index_of_biggest_in_corner += k;
319 
320  transpositions.coeffRef(k) = IndexType(index_of_biggest_in_corner);
321  if(k != index_of_biggest_in_corner)
322  {
323  // apply the transposition while taking care to consider only
324  // the lower triangular part
325  Index s = size-index_of_biggest_in_corner-1; // trailing size after the biggest element
326  mat.row(k).head(k).swap(mat.row(index_of_biggest_in_corner).head(k));
327  mat.col(k).tail(s).swap(mat.col(index_of_biggest_in_corner).tail(s));
328  std::swap(mat.coeffRef(k,k),mat.coeffRef(index_of_biggest_in_corner,index_of_biggest_in_corner));
329  for(Index i=k+1;i<index_of_biggest_in_corner;++i)
330  {
331  Scalar tmp = mat.coeffRef(i,k);
332  mat.coeffRef(i,k) = numext::conj(mat.coeffRef(index_of_biggest_in_corner,i));
333  mat.coeffRef(index_of_biggest_in_corner,i) = numext::conj(tmp);
334  }
335  if(NumTraits<Scalar>::IsComplex)
336  mat.coeffRef(index_of_biggest_in_corner,k) = numext::conj(mat.coeff(index_of_biggest_in_corner,k));
337  }
338 
339  // partition the matrix:
340  // A00 | - | -
341  // lu = A10 | A11 | -
342  // A20 | A21 | A22
343  Index rs = size - k - 1;
344  Block<MatrixType,Dynamic,1> A21(mat,k+1,k,rs,1);
345  Block<MatrixType,1,Dynamic> A10(mat,k,0,1,k);
346  Block<MatrixType,Dynamic,Dynamic> A20(mat,k+1,0,rs,k);
347 
348  if(k>0)
349  {
350  temp.head(k) = mat.diagonal().real().head(k).asDiagonal() * A10.adjoint();
351  mat.coeffRef(k,k) -= (A10 * temp.head(k)).value();
352  if(rs>0)
353  A21.noalias() -= A20 * temp.head(k);
354  }
355 
356  // In some previous versions of Eigen (e.g., 3.2.1), the scaling was omitted if the pivot
357  // was smaller than the cutoff value. However, since LDLT is not rank-revealing
358  // we should only make sure that we do not introduce INF or NaN values.
359  // Remark that LAPACK also uses 0 as the cutoff value.
360  RealScalar realAkk = numext::real(mat.coeffRef(k,k));
361  bool pivot_is_valid = (abs(realAkk) > RealScalar(0));
362 
363  if(k==0 && !pivot_is_valid)
364  {
365  // The entire diagonal is zero, there is nothing more to do
366  // except filling the transpositions, and checking whether the matrix is zero.
367  sign = ZeroSign;
368  for(Index j = 0; j<size; ++j)
369  {
370  transpositions.coeffRef(j) = IndexType(j);
371  ret = ret && (mat.col(j).tail(size-j-1).array()==Scalar(0)).all();
372  }
373  return ret;
374  }
375 
376  if((rs>0) && pivot_is_valid)
377  A21 /= realAkk;
378 
379  if(found_zero_pivot && pivot_is_valid) ret = false; // factorization failed
380  else if(!pivot_is_valid) found_zero_pivot = true;
381 
382  if (sign == PositiveSemiDef) {
383  if (realAkk < static_cast<RealScalar>(0)) sign = Indefinite;
384  } else if (sign == NegativeSemiDef) {
385  if (realAkk > static_cast<RealScalar>(0)) sign = Indefinite;
386  } else if (sign == ZeroSign) {
387  if (realAkk > static_cast<RealScalar>(0)) sign = PositiveSemiDef;
388  else if (realAkk < static_cast<RealScalar>(0)) sign = NegativeSemiDef;
389  }
390  }
391 
392  return ret;
393  }

395  // Reference for the algorithm: Davis and Hager, "Multiple Rank
396  // Modifications of a Sparse Cholesky Factorization" (Algorithm 1)
397  // Trivial rearrangements of their computations (Timothy E. Holy)
398  // allow their algorithm to work for rank-1 updates even if the
399  // original matrix is not of full rank.
400  // Here only rank-1 updates are implemented, to reduce the
401  // requirement for intermediate storage and improve accuracy
402  template<typename MatrixType, typename WDerived>
403  static bool updateInPlace(MatrixType& mat, MatrixBase<WDerived>& w, const typename MatrixType::RealScalar& sigma=1)
404  {
405  using numext::isfinite;
406  typedef typename MatrixType::Scalar Scalar;
407  typedef typename MatrixType::RealScalar RealScalar;
408 
409  const Index size = mat.rows();
410  eigen_assert(mat.cols() == size && w.size()==size);
411 
412  RealScalar alpha = 1;
413 
414  // Apply the update
415  for (Index j = 0; j < size; j++)
416  {
417  // Check for termination due to an original decomposition of low-rank
418  if (!(isfinite)(alpha))
419  break;
420 
421  // Update the diagonal terms
422  RealScalar dj = numext::real(mat.coeff(j,j));
423  Scalar wj = w.coeff(j);
424  RealScalar swj2 = sigma*numext::abs2(wj);
425  RealScalar gamma = dj*alpha + swj2;
426 
427  mat.coeffRef(j,j) += swj2/alpha;
428  alpha += swj2/dj;
429 
430 
431  // Update the terms of L
432  Index rs = size-j-1;
433  w.tail(rs) -= wj * mat.col(j).tail(rs);
434  if(gamma != 0)
435  mat.col(j).tail(rs) += (sigma*numext::conj(wj)/gamma)*w.tail(rs);
436  }
437  return true;
438  }

488 template<typename InputType>
489 LDLT<MatrixType,_UpLo>& LDLT<MatrixType,_UpLo>::compute(const EigenBase<InputType>& a)
490 {
491  check_template_parameters();
492 
493  eigen_assert(a.rows()==a.cols());
494  const Index size = a.rows();
495 
496  m_matrix = a.derived();
497 
498  // Compute matrix L1 norm = max abs column sum.
499  m_l1_norm = RealScalar(0);
500  // TODO move this code to SelfAdjointView
501  for (Index col = 0; col < size; ++col) {
502  RealScalar abs_col_sum;
503  if (_UpLo == Lower)
504  abs_col_sum = m_matrix.col(col).tail(size - col).template lpNorm<1>() + m_matrix.row(col).head(col).template lpNorm<1>();
505  else
506  abs_col_sum = m_matrix.col(col).head(col).template lpNorm<1>() + m_matrix.row(col).tail(size - col).template lpNorm<1>();
507  if (abs_col_sum > m_l1_norm)
508  m_l1_norm = abs_col_sum;
509  }
510 
511  m_transpositions.resize(size);
512  m_isInitialized = false;
513  m_temporary.resize(size);
514  m_sign = internal::ZeroSign;
515 
516  m_info = internal::ldlt_inplace<UpLo>::unblocked(m_matrix, m_transpositions, m_temporary, m_sign) ? Success : NumericalIssue;
517 
518  m_isInitialized = true;
519  return *this;
520 }

555 template<typename _MatrixType, int _UpLo>
556 template<typename RhsType, typename DstType>
557 void LDLT<_MatrixType,_UpLo>::_solve_impl(const RhsType &rhs, DstType &dst) const
558 {
559  eigen_assert(rhs.rows() == rows());
560  // dst = P b
561  dst = m_transpositions * rhs;
562 
563  // dst = L^-1 (P b)
564  matrixL().solveInPlace(dst);
565 
566  // dst = D^-1 (L^-1 P b)
567  // more precisely, use pseudo-inverse of D (see bug 241)
568  using std::abs;
569  const typename Diagonal<const MatrixType>::RealReturnType vecD(vectorD());
570  // In some previous versions, tolerance was set to the max of 1/highest and the maximal diagonal entry * epsilon
571  // as motivated by LAPACK's xGELSS:
572  // RealScalar tolerance = numext::maxi(vecD.array().abs().maxCoeff() * NumTraits<RealScalar>::epsilon(),RealScalar(1) / NumTraits<RealScalar>::highest());
573  // However, LDLT is not rank revealing, and so adjusting the tolerance wrt to the highest
574  // diagonal element is not well justified and leads to numerical issues in some cases.
575  // Moreover, Lapack's xSYTRS routines use 0 for the tolerance.
576  RealScalar tolerance = RealScalar(1) / NumTraits<RealScalar>::highest();
577 
578  for (Index i = 0; i < vecD.size(); ++i)
579  {
580  if(abs(vecD(i)) > tolerance)
581  dst.row(i) /= vecD(i);
582  else
583  dst.row(i).setZero();
584  }
585 
586  // dst = L^-T (D^-1 L^-1 P b)
587  matrixU().solveInPlace(dst);
588 
589  // dst = P^-1 (L^-T D^-1 L^-1 P b) = A^-1 b
590  dst = m_transpositions.transpose() * dst;
591 }
592 #endif
