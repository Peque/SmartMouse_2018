231 template<typename MatrixType, typename VectorType>
232 static Index llt_rank_update_lower(MatrixType& mat, const VectorType& vec, const typename MatrixType::RealScalar& sigma)
233 {
234  using std::sqrt;
235  typedef typename MatrixType::Scalar Scalar;
236  typedef typename MatrixType::RealScalar RealScalar;
237  typedef typename MatrixType::ColXpr ColXpr;
238  typedef typename internal::remove_all<ColXpr>::type ColXprCleaned;
239  typedef typename ColXprCleaned::SegmentReturnType ColXprSegment;
240  typedef Matrix<Scalar,Dynamic,1> TempVectorType;
241  typedef typename TempVectorType::SegmentReturnType TempVecSegment;
242 
243  Index n = mat.cols();
244  eigen_assert(mat.rows()==n && vec.size()==n);
245 
246  TempVectorType temp;
247 
248  if(sigma>0)
249  {
250  // This version is based on Givens rotations.
251  // It is faster than the other one below, but only works for updates,
252  // i.e., for sigma > 0
253  temp = sqrt(sigma) * vec;
254 
255  for(Index i=0; i<n; ++i)
256  {
257  JacobiRotation<Scalar> g;
258  g.makeGivens(mat(i,i), -temp(i), &mat(i,i));
259 
260  Index rs = n-i-1;
261  if(rs>0)
262  {
263  ColXprSegment x(mat.col(i).tail(rs));
264  TempVecSegment y(temp.tail(rs));
265  apply_rotation_in_the_plane(x, y, g);
266  }
267  }
268  }
269  else
270  {
271  temp = vec;
272  RealScalar beta = 1;
273  for(Index j=0; j<n; ++j)
274  {
275  RealScalar Ljj = numext::real(mat.coeff(j,j));
276  RealScalar dj = numext::abs2(Ljj);
277  Scalar wj = temp.coeff(j);
278  RealScalar swj2 = sigma*numext::abs2(wj);
279  RealScalar gamma = dj*beta + swj2;
280 
281  RealScalar x = dj + swj2/beta;
282  if (x<=RealScalar(0))
283  return j;
284  RealScalar nLjj = sqrt(x);
285  mat.coeffRef(j,j) = nLjj;
286  beta += swj2/dj;
287 
288  // Update the terms of L
289  Index rs = n-j-1;
290  if(rs)
291  {
292  temp.tail(rs) -= (wj/Ljj) * mat.col(j).tail(rs);
293  if(gamma != 0)
294  mat.col(j).tail(rs) = (nLjj/Ljj) * mat.col(j).tail(rs) + (nLjj * sigma*numext::conj(wj)/gamma)*temp.tail(rs);
295  }
296  }
297  }
298  return -1;
299 }

301 template<typename Scalar> struct llt_inplace<Scalar, Lower>
302 {
303  typedef typename NumTraits<Scalar>::Real RealScalar;
304  template<typename MatrixType>
305  static Index unblocked(MatrixType& mat)
306  {
307  using std::sqrt;
308 
309  eigen_assert(mat.rows()==mat.cols());
310  const Index size = mat.rows();
311  for(Index k = 0; k < size; ++k)
312  {
313  Index rs = size-k-1; // remaining size
314 
315  Block<MatrixType,Dynamic,1> A21(mat,k+1,k,rs,1);
316  Block<MatrixType,1,Dynamic> A10(mat,k,0,1,k);
317  Block<MatrixType,Dynamic,Dynamic> A20(mat,k+1,0,rs,k);
318 
319  RealScalar x = numext::real(mat.coeff(k,k));
320  if (k>0) x -= A10.squaredNorm();
321  if (x<=RealScalar(0))
322  return k;
323  mat.coeffRef(k,k) = x = sqrt(x);
324  if (k>0 && rs>0) A21.noalias() -= A20 * A10.adjoint();
325  if (rs>0) A21 /= x;
326  }
327  return -1;
328  }

330  template<typename MatrixType>
331  static Index blocked(MatrixType& m)
332  {
333  eigen_assert(m.rows()==m.cols());
334  Index size = m.rows();
335  if(size<32)
336  return unblocked(m);
337 
338  Index blockSize = size/8;
339  blockSize = (blockSize/16)*16;
340  blockSize = (std::min)((std::max)(blockSize,Index(8)), Index(128));
341 
342  for (Index k=0; k<size; k+=blockSize)
343  {
344  // partition the matrix:
345  // A00 | - | -
346  // lu = A10 | A11 | -
347  // A20 | A21 | A22
348  Index bs = (std::min)(blockSize, size-k);
349  Index rs = size - k - bs;
350  Block<MatrixType,Dynamic,Dynamic> A11(m,k, k, bs,bs);
351  Block<MatrixType,Dynamic,Dynamic> A21(m,k+bs,k, rs,bs);
352  Block<MatrixType,Dynamic,Dynamic> A22(m,k+bs,k+bs,rs,rs);
353 
354  Index ret;
355  if((ret=unblocked(A11))>=0) return k+ret;
356  if(rs>0) A11.adjoint().template triangularView<Upper>().template solveInPlace<OnTheRight>(A21);
357  if(rs>0) A22.template selfadjointView<Lower>().rankUpdate(A21,typename NumTraits<RealScalar>::Literal(-1)); // bottleneck
358  }
359  return -1;
360  }

424 LLT<MatrixType,_UpLo>& LLT<MatrixType,_UpLo>::compute(const EigenBase<InputType>& a)
425 {
426  check_template_parameters();
427 
428  eigen_assert(a.rows()==a.cols());
429  const Index size = a.rows();
430  m_matrix.resize(size, size);
431  if (!internal::is_same_dense(m_matrix, a.derived()))
432  m_matrix = a.derived();
433 
434  // Compute matrix L1 norm = max abs column sum.
435  m_l1_norm = RealScalar(0);
436  // TODO move this code to SelfAdjointView
437  for (Index col = 0; col < size; ++col) {
438  RealScalar abs_col_sum;
439  if (_UpLo == Lower)
440  abs_col_sum = m_matrix.col(col).tail(size - col).template lpNorm<1>() + m_matrix.row(col).head(col).template lpNorm<1>();
441  else
442  abs_col_sum = m_matrix.col(col).head(col).template lpNorm<1>() + m_matrix.row(col).tail(size - col).template lpNorm<1>();
443  if (abs_col_sum > m_l1_norm)
444  m_l1_norm = abs_col_sum;
445  }
446 
447  m_isInitialized = true;
448  bool ok = Traits::inplace_decomposition(m_matrix);
449  m_info = ok ? Success : NumericalIssue;
450 
451  return *this;
452 }
