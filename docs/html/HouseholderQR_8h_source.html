254 template<typename MatrixQR, typename HCoeffs>
255 void householder_qr_inplace_unblocked(MatrixQR& mat, HCoeffs& hCoeffs, typename MatrixQR::Scalar* tempData = 0)
256 {
257  typedef typename MatrixQR::Scalar Scalar;
258  typedef typename MatrixQR::RealScalar RealScalar;
259  Index rows = mat.rows();
260  Index cols = mat.cols();
261  Index size = (std::min)(rows,cols);
262 
263  eigen_assert(hCoeffs.size() == size);
264 
265  typedef Matrix<Scalar,MatrixQR::ColsAtCompileTime,1> TempType;
266  TempType tempVector;
267  if(tempData==0)
268  {
269  tempVector.resize(cols);
270  tempData = tempVector.data();
271  }
272 
273  for(Index k = 0; k < size; ++k)
274  {
275  Index remainingRows = rows - k;
276  Index remainingCols = cols - k - 1;
277 
278  RealScalar beta;
279  mat.col(k).tail(remainingRows).makeHouseholderInPlace(hCoeffs.coeffRef(k), beta);
280  mat.coeffRef(k,k) = beta;
281 
282  // apply H to remaining part of m_qr from the left
283  mat.bottomRightCorner(remainingRows, remainingCols)
284  .applyHouseholderOnTheLeft(mat.col(k).tail(remainingRows-1), hCoeffs.coeffRef(k), tempData+k+1);
285  }
286 }

289 template<typename MatrixQR, typename HCoeffs,
290  typename MatrixQRScalar = typename MatrixQR::Scalar,
291  bool InnerStrideIsOne = (MatrixQR::InnerStrideAtCompileTime == 1 && HCoeffs::InnerStrideAtCompileTime == 1)>
292 struct householder_qr_inplace_blocked
293 {
294  // This is specialized for MKL-supported Scalar types in HouseholderQR_MKL.h
295  static void run(MatrixQR& mat, HCoeffs& hCoeffs, Index maxBlockSize=32,
296  typename MatrixQR::Scalar* tempData = 0)
297  {
298  typedef typename MatrixQR::Scalar Scalar;
299  typedef Block<MatrixQR,Dynamic,Dynamic> BlockType;
300 
301  Index rows = mat.rows();
302  Index cols = mat.cols();
303  Index size = (std::min)(rows, cols);
304 
305  typedef Matrix<Scalar,Dynamic,1,ColMajor,MatrixQR::MaxColsAtCompileTime,1> TempType;
306  TempType tempVector;
307  if(tempData==0)
308  {
309  tempVector.resize(cols);
310  tempData = tempVector.data();
311  }
312 
313  Index blockSize = (std::min)(maxBlockSize,size);
314 
315  Index k = 0;
316  for (k = 0; k < size; k += blockSize)
317  {
318  Index bs = (std::min)(size-k,blockSize); // actual size of the block
319  Index tcols = cols - k - bs; // trailing columns
320  Index brows = rows-k; // rows of the block
321 
322  // partition the matrix:
323  // A00 | A01 | A02
324  // mat = A10 | A11 | A12
325  // A20 | A21 | A22
326  // and performs the qr dec of [A11^T A12^T]^T
327  // and update [A21^T A22^T]^T using level 3 operations.
328  // Finally, the algorithm continue on A22
329 
330  BlockType A11_21 = mat.block(k,k,brows,bs);
331  Block<HCoeffs,Dynamic,1> hCoeffsSegment = hCoeffs.segment(k,bs);
332 
333  householder_qr_inplace_unblocked(A11_21, hCoeffsSegment, tempData);
334 
335  if(tcols)
336  {
337  BlockType A21_22 = mat.block(k,k+bs,brows,tcols);
338  apply_block_householder_on_the_left(A21_22,A11_21,hCoeffsSegment, false); // false == backward
339  }
340  }
341  }
342 };

347 template<typename _MatrixType>
348 template<typename RhsType, typename DstType>
349 void HouseholderQR<_MatrixType>::_solve_impl(const RhsType &rhs, DstType &dst) const
350 {
351  const Index rank = (std::min)(rows(), cols());
352  eigen_assert(rhs.rows() == rows());
353 
354  typename RhsType::PlainObject c(rhs);
355 
356  // Note that the matrix Q = H_0^* H_1^*... so its inverse is Q^* = (H_0 H_1 ...)^T
357  c.applyOnTheLeft(householderSequence(
358  m_qr.leftCols(rank),
359  m_hCoeffs.head(rank)).transpose()
360  );
361 
362  m_qr.topLeftCorner(rank, rank)
363  .template triangularView<Upper>()
364  .solveInPlace(c.topRows(rank));
365 
366  dst.topRows(rank) = c.topRows(rank);
367  dst.bottomRows(cols()-rank).setZero();
368 }
